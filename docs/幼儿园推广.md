# 幼儿园推广的接口文档

## 渠道管理列表（包含搜索）

```
{
	"url": "http://{ip:port}/channel/channel/list",
	
	"http-method": "get",
	
	"request-params": {
            "name":"xxx", # 渠道名称，非必填
            "contact_name":"xxx", # 联系人，非必填
            "contact_mobile":"xxx" # 联系电话，非必填
            "page": "1", # 第几页，必填
            "rows": "20", # 一页几条，必填
	},
	
	"response-data": {
            "data": {
                "count":"20",
                "list":[
                    {
                        "id":"xxx", # 渠道唯一id
                        "name":"xxx", # 渠道名称
                        "username":"xxx", # 账号
                        "contact_name":"xxx", # 联系人
                        "contact_mobile":"xxx", # 联系电话
                        "remark":"xxx", # 备注
                        "create_time":"xxx", # 创建时间
                    },
                    {
                        "id":"xxx", # 渠道唯一id
                        "name":"xxx", # 渠道名称
                        "username":"xxx", # 账号
                        "contact_name":"xxx", # 联系人
                        "contact_mobile":"xxx", # 联系电话
                        "remark":"xxx", # 备注
                        "create_time":"xxx", # 创建时间
                    },
                ]
            },
            "resultCode": 0,
            "resultMsg": "操作成功"
        }
	
}
```

## 新建渠道

```
{
	"url": "http://{ip:port}/channel/channel/add",
	
	"http-method": "post",
	
	"request-params": {
            "name":"xxx", # 渠道名称，必填
            "username":"xxx", # 账号，必填
            "contact_name":"xxx", # 联系人，必填
            "contact_mobile":"xxx" # 联系电话，必填
            "remark":"xxx" # 备注，非必填
	},
	
	"response-data": {
            "data": {
                "url":"xxx", # 渠道登录地址
                "username":"xxx", # 账号
                "password":"123456" # 登录密码，默认为123456
            },
            "resultCode": 0,
            "resultMsg": "操作成功"
        }
	
}
```
