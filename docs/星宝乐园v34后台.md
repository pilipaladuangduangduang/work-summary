## 用户列表

```
{
    "url": "http://{ip:port}/cms/user/list",

    "http-method": "get",

    "request-params": {
        "page_num":"页数",
        "rows":"每页的条数"
    },

    "response-data": {
        "result": {
	    "list": [
	       {
	           "id":"用户ID",
		   "username":"账户名称",
	           "nickname":"用户昵称",
	           "role_name":"角色名称",
	           "desc":"用户简介",
		   "role":"角色ID",
		   "publisher":"合作方ID",
		   "publisher_name":"合作方名称"
	       }
	        ...
	    ],
	    "count": 1
        },

}
```
